📱 ИНСТРУКЦИЯ: КОМПИЛЯЦИЯ В APK (Android)
═════════════════════════════════════════════

✅ ТРЕБОВАНИЯ:
━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━
- Python 3.7+
- Java JDK 11+ (обязательно!)
- Android SDK
- Android NDK


📋 ШАГ 1: УСТАНОВКА BUILDOZER
━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━

pip install buildozer
pip install cython
pip install kivy
pip install requests


📋 ШАГ 2: ПРОВЕРКА JAVA
━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━

java -version
javac -version


Если Java не установлена:
https://www.oracle.com/java/technologies/downloads/

ДЛЯ WINDOWS (chocolatey):
choco install openjdk11


📋 ШАГ 3: ПЕРЕЙТИ В ПАПКУ ПРОЕКТА
━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━

cd "C:\Users\Артём\Desktop\папка Артёма\PYTON\браузер\браузер релиз\"


📋 ШАГ 4: ЗАПУСТИТЬ КОМПИЛЯЦИЮ
━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━

buildozer android debug

⏳ Это займет 5-15 минут в первый раз!


✅ ГОТОВЫЙ APK БУДЕТ В:
━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━

./bin/nexusbrowser-1.0-debug.apk


🔧 ЕСЛИ ВОЗНИКНУТ ОШИБКИ:
━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━

1. Очистка кеша:
   buildozer android debug -- --clear_cache

2. Полная очистка:
   rm -r .buildozer
   buildozer android debug

3. Установить Gradle (если ошибка с Gradle):
   pip install gradle

4. Проверить переменные окружения (ANDROID_HOME, JAVA_HOME)


📱 УСТАНОВКА НА ТЕЛЕФОН
━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━

1. Включить "Разработку" на телефоне
2. Подключить телефон USB кабелем
3. Выполнить:

   adb install bin/nexusbrowser-1.0-debug.apk

Или загрузить файл вручную на телефон и установить.
